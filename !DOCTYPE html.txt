    <script>
        // --- SISTEMA DE VERSIONES ANTI-CACHÉ ---
        const STORE_VERSION_TIMESTAMP = 1718000000000; 

        // --- BASE DE DATOS DE PRODUCTOS ---
        /* START_PRODUCTS_ARRAY */
        const initialProducts = [
            { id: 1, name: "PRENDA DE PRUEBA", price: 50, img: "IMG_1846.jpeg" }
        ];
        /* END_PRODUCTS_ARRAY */

        let products = [...initialProducts];

        // Función para dibujar los productos en la pantalla
        function renderCatalogo() {
            const catalogo = document.getElementById('catalogo');
            catalogo.innerHTML = "";

            products.forEach(p => {
                catalogo.innerHTML += `
                    <div class="producto">
                        <div class="img-container">
                            <img src="${p.img}" alt="${p.name}">
                        </div>
                        <h3>${p.name}</h3>
                        <div class="precio">$${p.price}</div>
                        <button class="btn-compra">AÑADIR AL CARRITO</button>
                    </div>
                `;
            });
        }

        // Seguridad Básica del Panel
        function accederPanel() {
            let pass = prompt("Contraseña de Acceso:");
            if (pass === "reupvz123") {
                document.getElementById('admin-panel').style.display = 'flex';
            } else if (pass !== null) {
                alert("Acceso Denegado 🛑");
            }
        }

        function cerrarPanel() {
            document.getElementById('admin-panel').style.display = 'none';
        }

        // Agregar nueva prenda
        function agregarProducto() {
            const name = document.getElementById('prod-name').value;
            const price = document.getElementById('prod-price').value;
            let img = document.getElementById('prod-img').value;

            if (!name || !price) {
                alert("Faltan datos de la prenda.");
                return;
            }

            if(!img) img = "IMG_1846.jpeg";

            const newId = Date.now();
            products.push({ id: newId, name: name, price: price, img: img });
            
            renderCatalogo();
            alert("Prenda añadida al catálogo. ¡No olvides descargar el HTML para guardar!");
            
            // Limpiar inputs
            document.getElementById('prod-name').value = "";
            document.getElementById('prod-price').value = "";
        }

        // --- SISTEMA DE HORNEADO (BAKING) Y ANTI-CACHÉ ---
        function downloadUpdatedHTML() {
            // 1. Ocultar panel antes de "tomar la foto" del código
            cerrarPanel();

            // 2. Extraer todo el HTML actual de la página
            let htmlContent = "<!DOCTYPE html>\n" + document.documentElement.outerHTML;

            // 3. Generar un nuevo número de serie anti-caché
            const newVersion = Date.now();
            htmlContent = htmlContent.replace(/const STORE_VERSION_TIMESTAMP = \d+;/, `const STORE_VERSION_TIMESTAMP = ${newVersion};`);

            // 4. Inyectar el inventario actual permanentemente en el código fuente
            const productsJSON = JSON.stringify(products, null, 4);
            const regex = /\/\* START_PRODUCTS_ARRAY \*\/[\s\S]*?\/\* END_PRODUCTS_ARRAY \*\//;
            htmlContent = htmlContent.replace(regex, `/* START_PRODUCTS_ARRAY */\n        const initialProducts = ${productsJSON};\n        /* END_PRODUCTS_ARRAY */`);

            // 5. Descargar el archivo
            const blob = new Blob([htmlContent], { type: 'text/html' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'index.html';
            a.click();
            URL.revokeObjectURL(url);
        }

        // Inicializar la página
        window.onload = function() {
            renderCatalogo();
        };
    </script>
